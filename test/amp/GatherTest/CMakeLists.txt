# List the names of common files to compile across all platforms
set( ampBolt.Test.Gather.Source  GatherTest.cpp )

set( ampBolt.Test.Gather.Headers   ${BOLT_INCLUDE_DIR}/bolt/amp/gather.h
                                   ${BOLT_INCLUDE_DIR}/bolt/amp/detail/gather.inl )

set( ampBolt.Test.Gather.Files ${ampBolt.Test.Gather.Source} ${ampBolt.Test.Gather.Headers} )

# Set project specific compile and link options
if( MSVC )
    set( CMAKE_CXX_FLAGS "-bigobj ${CMAKE_CXX_FLAGS}" )
    set( CMAKE_C_FLAGS "-bigobj ${CMAKE_C_FLAGS}" )
endif()

add_executable( ampBolt.Test.Gather ${ampBolt.Test.Gather.Files} )

if(BUILD_TBB)
    target_link_libraries( ampBolt.Test.Gather ampBolt.Runtime ${GTEST_LIBRARIES} ${Boost_LIBRARIES}  ${TBB_LIBRARIES} )
else (BUILD_TBB)
    target_link_libraries( ampBolt.Test.Gather ampBolt.Runtime ${GTEST_LIBRARIES} ${Boost_LIBRARIES}  )
endif()

set_target_properties( ampBolt.Test.Gather PROPERTIES VERSION ${Bolt_VERSION} )
set_target_properties( ampBolt.Test.Gather PROPERTIES RUNTIME_OUTPUT_DIRECTORY "${PROJECT_BINARY_DIR}/staging" )

set_property( TARGET ampBolt.Test.Gather PROPERTY FOLDER "Test/AMP")
        
# CPack configuration; include the executable into the package
install( TARGETS ampBolt.Test.Gather
    RUNTIME DESTINATION ${BIN_DIR}
    LIBRARY DESTINATION ${LIB_DIR}
    ARCHIVE DESTINATION ${LIB_DIR}/import
    )

install( FILES       
         )

install( FILES       
         )
